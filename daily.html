function loadReport() { const date = document.getElementById('dateFilter').value; const body = document.getElementById('body'); body.innerHTML = "جاري الفحص..."; db.collection("attendance").get().then((snap) => { body.innerHTML = ""; if (snap.empty) { body.innerHTML = "قاعدة البيانات فارغة تماماً!"; return; } let found = false; snap.forEach((doc) => { // لنطبع في الجدول كل معرفات الوثائق لنرى صيغتها body.innerHTML += `ID الموجود: ${doc.id}`; if(doc.id.startsWith(date)) { found = true; const data = doc.data().data; const parts = doc.id.split('_'); for(let sId in data) { if(data[sId] !== "حاضر") { body.innerHTML += `${sId}${parts[1]}${data[sId]}`; } } } }); if(!found) { body.innerHTML += "لا توجد تطابقات للتاريخ المحدد!"; } }).catch(err => { body.innerHTML = "خطأ: " + err.message + ""; }); }
