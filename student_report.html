async function generateReport() { const sId = document.getElementById('studentSelect').value; const dFrom = document.getElementById('dateFrom').value; const dTo = document.getElementById('dateTo').value; const body = document.getElementById('reportBody'); const summary = document.getElementById('summary'); if (!sId || !dFrom || !dTo) { alert("يرجى التأكد من اختيار الطالب وتحديد تاريخ البداية والنهاية"); return; } body.innerHTML = "جاري البحث..."; summary.innerHTML = ""; try { // البحث فقط في سجلات الحضور التي تخص هذا الطالب // ملاحظة: إذا كان هيكل بياناتك في attendance يعتمد على اسم المستند كالتاريخ // فيجب جلبها وتصفيتها برمجياً، لكن سنحسن الأداء: const attendSnap = await db.collection("attendance").get(); let abs = 0, leave = 0, rows = ""; attendSnap.forEach(doc => { const docId = doc.id; // مثال: 2026-06-27_الرابع العلمي_أ_الدرس الأول const parts = docId.split('_'); const date = parts[0]; // تحقق إذا كان التاريخ ضمن الفترة المطلوبة if (date >= dFrom && date <= dTo) { const data = doc.data().data || {}; // تحقق إذا كان الطالب موجوداً في هذا السجل if (data[sId]) { const status = data[sId]; if (status !== "حاضر") { if (status === "غائب") abs++; if (status === "مجاز") leave++; rows += ` ${date} ${parts[3] || 'غير محدد'} ${status} `; } } } }); body.innerHTML = rows || "لا توجد سجلات غياب أو إجازة لهذا الطالب في هذه الفترة"; summary.innerHTML = `إجمالي الغياب: ${(abs / 4).toFixed(1)} يوم | الإجازات: ${(leave / 4).toFixed(1)} يوم`; } catch (error) { console.error("خطأ في جلب البيانات:", error); body.innerHTML = "حدث خطأ أثناء جلب البيانات، تأكد من اتصال الإنترنت."; } }
